### 1.2 实体清单

## 2. PostgreSQL数据库Schema

### 2.1 用户相关表

#### users（用户表）

```sql
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    phone           VARCHAR(20) UNIQUE NOT NULL,
    nickname        VARCHAR(50),
    avatar_url      VARCHAR(500),
    
    -- 用户配置
    default_reminder_level VARCHAR(20) DEFAULT 'standard',  -- 温和/标准/重要/强制
    default_remind_offset  INTEGER DEFAULT 5,               -- 默认提前提醒分钟数
    default_repeat_interval INTEGER DEFAULT 10,             -- 默认重复间隔
    default_max_reminders   INTEGER DEFAULT 3,              -- 默认最大提醒次数
    allow_snooze_default    BOOLEAN DEFAULT TRUE,           -- 默认允许延后
    allow_skip_default      BOOLEAN DEFAULT TRUE,           -- 默认允许跳过
    quiet_hours_start       TIME DEFAULT '22:00',           -- 免打扰开始时间
    quiet_hours_end         TIME DEFAULT '08:00',           -- 免打扰结束时间
    
    -- AI服务配置
    preferred_asr_provider  VARCHAR(20) DEFAULT 'xunfei',   -- 讯飞/阿里云
    preferred_nlp_provider  VARCHAR(20) DEFAULT 'xunfei',
    
    -- 统计字段
    total_records           INTEGER DEFAULT 0,              -- 总记录数
    total_agendas           INTEGER DEFAULT 0,              -- 总事程数
    match_rate              DECIMAL(5,2) DEFAULT 0,         -- 匹配率
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    last_active_at  TIMESTAMP WITH TIME ZONE,
    
    -- 状态
    is_active       BOOLEAN DEFAULT TRUE,
    is_deleted      BOOLEAN DEFAULT FALSE
);

-- 索引
CREATE INDEX idx_users_phone ON users(phone);
CREATE INDEX idx_users_last_active ON users(last_active_at);
```

#### family_bindings（家属绑定表）

```sql
CREATE TABLE family_bindings (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    family_user_id  UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 权限级别
    permission_level VARCHAR(20) NOT NULL DEFAULT 'viewer',  -- viewer/confirmer/admin
    
    -- 绑定状态
    status          VARCHAR(20) DEFAULT 'pending',  -- pending/active/disabled
    
    -- 通知配置
    notify_on_missed    BOOLEAN DEFAULT TRUE,       -- 遗漏时通知家属
    notify_on_forced    BOOLEAN DEFAULT TRUE,       -- 强制事程未完成时通知
    notify_offset_minutes INTEGER DEFAULT 30,       -- 遗漏多久后通知
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    accepted_at     TIMESTAMP WITH TIME ZONE,
    
    -- 约束：一个用户不能重复绑定同一个家属
    CONSTRAINT unique_family_binding UNIQUE (user_id, family_user_id)
);

-- 紦引
CREATE INDEX idx_family_bindings_user ON family_bindings(user_id);
CREATE INDEX idx_family_bindings_family ON family_bindings(family_user_id);
```

### 2.2 时间线记录表

#### timeline_records（时间线记录表）

```sql
CREATE TABLE timeline_records (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 核心内容
    timestamp       TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    content         TEXT NOT NULL,                          -- 原始语音/文字内容
    behavior_tag    VARCHAR(100),                           -- 提取的行为标签
    
    -- 语音信息
    voice_file_url  VARCHAR(500),                           -- 语音文件OSS地址
    voice_duration  INTEGER,                                -- 语音时长（秒）
    asr_provider    VARCHAR(20),                            -- ASR服务提供商
    asr_raw_result  JSONB,                                  -- ASR原始返回结果
    
    -- NLP信息
    nlp_provider    VARCHAR(20),                            -- NLP服务提供商
    nlp_intent      VARCHAR(50),                            -- 意图类型（record/plan/query）
    nlp_time_expr   VARCHAR(100),                           -- 时间表达式
    nlp_status_expr VARCHAR(50),                            -- 状态表达式（正在/完成/准备）
    nlp_entities    JSONB,                                  -- NLP提取的其他实体
    
    -- 匹配信息
    matched_agenda_id   UUID REFERENCES agendas(id),        -- 匹配到的事程ID
    match_score         DECIMAL(5,2),                       -- 匹配分数（0-1）
    match_method        VARCHAR(20),                        -- 匹配方式（auto/manual）
    matched_at          TIMESTAMP WITH TIME ZONE,           -- 匹配时间
    
    -- 来源信息
    source          VARCHAR(20) DEFAULT 'voice',            -- voice/manual/import
    device_id       VARCHAR(100),                           -- 设备标识
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 状态
    is_deleted      BOOLEAN DEFAULT FALSE
);

-- 紦索引
CREATE INDEX idx_timeline_user_date ON timeline_records(user_id, timestamp DESC);
CREATE INDEX idx_timeline_behavior ON timeline_records(behavior_tag);
CREATE INDEX idx_timeline_matched ON timeline_records(matched_agenda_id) WHERE matched_agenda_id IS NOT NULL;
CREATE INDEX idx_timeline_timestamp ON timeline_records(timestamp DESC);
```

### 2.3 计划事程表

#### agendas（计划事程表）

```sql
CREATE TABLE agendas (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 核心内容
    planned_time    TIMESTAMP WITH TIME ZONE NOT NULL,      -- 计划执行时间
    content         TEXT NOT NULL,                          -- 事程内容
    behavior_tag    VARCHAR(100),                           -- 行为标签（用于匹配）
    
    -- 事程类型
    agenda_type     VARCHAR(20) DEFAULT 'fixed',            -- fixed/floating/pure_record
    -- fixed: 固定时间事程
    -- floating: 浮动事程（无固定时间，状态感知触发）
    -- pure_record: 纯记录模式（不提醒）
    
    -- 浮动事程配置（agenda_type = 'floating'时生效）
    floating_time_range_start TIME,                         -- 可接受时间范围开始
    floating_time_range_end   TIME,                         -- 可接受时间范围结束
    floating_trigger_conditions JSONB,                      -- 触发条件配置
    
    -- 状态
    status          VARCHAR(20) DEFAULT 'pending',          -- pending/active/matched/skipped/expired
    -- pending: 待开始（计划时间未到）
    -- active: 待验证（计划时间已到，等待用户确认）
    -- matched: 已匹配（用户已完成）
    -- skipped: 已跳过（用户主动跳过或超时）
    -- expired: 已过期（超过时间窗口）
    
    -- 完成信息
    matched_timeline_id UUID REFERENCES timeline_records(id), -- 匹配的时间线记录
    matched_at      TIMESTAMP WITH TIME ZONE,               -- 匹配时间
    skipped_at      TIMESTAMP WITH TIME ZONE,               -- 跳过时间
    skip_reason     VARCHAR(100),                           -- 跳过原因
    
    -- 提醒配置（继承用户默认值，可覆盖）
    remind_offset   INTEGER DEFAULT 5,                      -- 提前提醒分钟数
    remind_level    VARCHAR(20) DEFAULT 'standard',         -- 温和/标准/重要/强制
    
    -- 重复配置
    is_recurring    BOOLEAN DEFAULT FALSE,                  -- 是否重复事程
    recurring_rule  VARCHAR(100),                           -- 重复规则（RRULE格式）
    recurring_end_date DATE,                                -- 重复结束日期
    parent_agenda_id UUID REFERENCES agendas(id),           -- 父事程ID（重复事程）
    
    -- 强制事程配置（remind_level = 'forced'时生效）
    is_forced       BOOLEAN DEFAULT FALSE,
    forced_strategy_id UUID REFERENCES escalating_strategies(id),
    family_notify_on_miss BOOLEAN DEFAULT TRUE,             -- 遗漏时通知家属
    
    -- 来源
    source          VARCHAR(20) DEFAULT 'manual',           -- manual/auto_extracted/voice_created
    auto_extract_confidence DECIMAL(5,2),                   -- 自动提取置信度
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 状态
    is_deleted      BOOLEAN DEFAULT FALSE
);

-- 紦索引
CREATE INDEX idx_agendas_user_date ON agendas(user_id, planned_time);
CREATE INDEX idx_agendas_status ON agendas(status);
CREATE INDEX idx_agendas_behavior ON agendas(behavior_tag);
CREATE INDEX idx_agendas_recurring ON agendas(parent_agenda_id) WHERE parent_agenda_id IS NOT NULL;
CREATE INDEX idx_agendas_active ON agendas(user_id, planned_time, status) 
    WHERE status IN ('pending', 'active') AND is_deleted = FALSE;
```

### 2.4 提醒规则表

#### reminder_rules（提醒规则表）

```sql
CREATE TABLE reminder_rules (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    agenda_id       UUID REFERENCES agendas(id) ON DELETE CASCADE,  -- NULL表示全局默认规则
    
    -- 提醒级别
    level           VARCHAR(20) NOT NULL DEFAULT 'standard', -- gentle/standard/important/forced
    
    -- 提醒方式开关
    enable_notification  BOOLEAN DEFAULT TRUE,              -- 通知栏
    enable_sound         BOOLEAN DEFAULT TRUE,              -- 声音
    enable_vibration     BOOLEAN DEFAULT TRUE,              -- 震动
    enable_popup         BOOLEAN DEFAULT FALSE,             -- 弹窗
    enable_voice         BOOLEAN DEFAULT FALSE,             -- 语音播报
    
    -- 提醒强度配置
    vibration_strength   VARCHAR(20) DEFAULT 'medium',      -- weak/medium/strong
    sound_volume         INTEGER DEFAULT 80,                -- 音量百分比
    
    -- 时间配置
    remind_offset        INTEGER DEFAULT 5,                 -- 提前提醒分钟数
    repeat_interval      INTEGER DEFAULT 10,                -- 重复提醒间隔
    max_reminders        INTEGER DEFAULT 3,                 -- 最大提醒次数
    
    -- 用户操作权限
    allow_snooze         BOOLEAN DEFAULT TRUE,              -- 允许延后
    allow_skip           BOOLEAN DEFAULT TRUE,              -- 允许跳过
    snooze_options       INTEGER[] DEFAULT ARRAY[5,10,15,30], -- 延后选项（分钟）
    
    -- 强制级别配置（level = 'forced'时生效）
    escalating_strategy_id UUID REFERENCES escalating_strategies(id),
    family_notify_offset  INTEGER DEFAULT 30,               -- 家属通知延迟（分钟）
    
    -- 条件触发配置
    enable_geo_trigger    BOOLEAN DEFAULT FALSE,            -- 地理位置触发
    geo_location          JSONB,                            -- {lat, lng, radius, type}
    enable_weather_trigger BOOLEAN DEFAULT FALSE,           -- 天气触发
    weather_conditions    VARCHAR(50)[] DEFAULT ARRAY[],    -- rainy/hot/foggy
    enable_behavior_chain BOOLEAN DEFAULT FALSE,            -- 行为链触发
    behavior_chain_config JSONB,                            -- {pre_behavior, delay_minutes}
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 约束：一个事程只能有一个提醒规则
    CONSTRAINT unique_agenda_reminder_rule UNIQUE (agenda_id) WHERE agenda_id IS NOT NULL
);

-- 紦索引
CREATE INDEX idx_reminder_rules_user ON reminder_rules(user_id);
CREATE INDEX idx_reminder_rules_agenda ON reminder_rules(agenda_id);
```

### 2.6 匹配结果表

#### match_results（匹配结果表）

```sql
CREATE TABLE match_results (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 关联记录
    timeline_id     UUID NOT NULL REFERENCES timeline_records(id) ON DELETE CASCADE,
    agenda_id       UUID NOT NULL REFERENCES agendas(id) ON DELETE CASCADE,
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 匹配信息
    match_score     DECIMAL(5,2) NOT NULL,                  -- 匹配分数（0-1）
    match_method    VARCHAR(20) NOT NULL,                   -- keyword/time_window/manual
    
    -- 匹配详情
    keyword_match_score  DECIMAL(3,2),                      -- 关键词匹配分数
    time_window_match    BOOLEAN,                          -- 时间窗口是否匹配
    time_deviation_minutes INTEGER,                        -- 时间偏差分钟数
    
    -- 匹配状态
    status          VARCHAR(20) DEFAULT 'auto_matched',     -- auto_matched/user_confirmed/user_rejected
    
    -- 时间戳
    matched_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    confirmed_at    TIMESTAMP WITH TIME ZONE,               -- 用户确认时间（如需确认）
    
    -- 反馈（用于优化匹配算法）
    user_feedback   VARCHAR(20),                           -- correct/incorrect
    feedback_reason TEXT                                   -- 用户反馈原因
);

-- 紦索引
CREATE INDEX idx_match_results_timeline ON match_results(timeline_id);
CREATE INDEX idx_match_results_agenda ON match_results(agenda_id);
CREATE INDEX idx_match_results_user ON match_results(user_id);
CREATE INDEX idx_match_results_time ON match_results(matched_at DESC);
```

### 2.7 分析报告表

#### analytics_reports（分析报告表）

```sql
CREATE TABLE analytics_reports (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 报告周期
    report_type     VARCHAR(20) NOT NULL,                   -- daily/weekly/monthly
    period_start    DATE NOT NULL,
    period_end      DATE NOT NULL,
    
    -- 统计数据
    total_records   INTEGER DEFAULT 0,                      -- 总记录数
    total_agendas   INTEGER DEFAULT 0,                      -- 总事程数
    matched_count   INTEGER DEFAULT 0,                      -- 匹配成功数
    skipped_count   INTEGER DEFAULT 0,                      -- 跳过数
    match_rate      DECIMAL(5,2) DEFAULT 0,                 -- 匹配率
    
    -- 行为频率分析
    behavior_frequency JSONB DEFAULT '{}'::jsonb,           -- {behavior_tag: count}
    
    -- 时间分析
    avg_wake_time   TIME,                                   -- 平均起床时间
    avg_sleep_time  TIME,                                   -- 平均睡觉时间
    active_hours    DECIMAL(4,2),                           -- 活跃时长（小时）
    
    -- 异常检测
    anomalies       JSONB DEFAULT '[]'::jsonb,              -- 异常列表
    
    -- 洞察建议
    insights        JSONB DEFAULT '[]'::jsonb,              -- 洞察建议列表
    
    -- 时间戳
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

### 4.2 同步流程

#### 4.2.1 本地创建 → 云端同步

```dart
// 1. 本地创建记录
final record = TimelineRecord(
  id: generateUUID(),
  userId: currentUser.id,
  timestamp: DateTime.now(),

  behaviorTag: '吃药',

// 2. 加入同步队列

  type: SyncType.create,
  entity: 'timeline_record',
  data: record.toJson(),
  priority: SyncPriority.high,
));

// 3. 后台同步服务处理
class BackgroundSyncService {
  Future<void> processSyncQueue() async {
    final pendingTasks = await syncQueue.getPendingTasks();
    
    for (final task in pendingTasks) {
      try {
        // 更新状态为syncing
        await syncQueue.updateStatus(task.id, SyncStatus.syncing);
        
        // 调用API
        final response = await apiClient.post('/timeline', task.data);
        
        if (response.success) {
          // 更新本地记录的云端ID和时间戳
          await localRepo.updateSyncInfo(task.localId, response.data);
          await syncQueue.updateStatus(task.id, SyncStatus.synced);
        } else {
          await syncQueue.updateStatus(task.id, SyncStatus.failed, response.error);
        }
      } catch (e) {
        await syncQueue.updateStatus(task.id, SyncStatus.failed, e.toString());
      }
    }
  }
}
```

#### 4.2.2 云端推送 → 本地更新

```dart
// WebSocket接收云端推送
class WebSocketHandler {
  void handleMessage(dynamic message) {
    final data = jsonDecode(message);
    
    switch (data['type']) {
      case 'agenda_updated':
        _handleAgendaUpdate(data['payload']);
        break;
      case 'timeline_matched':
        _handleTimelineMatched(data['payload']);
        break;
      case 'reminder_triggered':
        _handleReminderTriggered(data['payload']);
        break;
    }
  }
  
  Future<void> _handleAgendaUpdate(Map<String, dynamic> payload) async {
    final agenda = Agenda.fromJson(payload);
    
    // 检查本地是否存在
    final localAgenda = await localAgendaRepo.getById(agenda.id);
    
    if (localAgenda == null) {
      // 本地不存在，直接插入
      await localAgendaRepo.insert(agenda.copyWith(syncStatus: SyncStatus.synced));
    } else {
      // 本地存在，比较时间戳决定是否更新
      if (agenda.updatedAt.isAfter(localAgenda.updatedAt)) {
        await localAgendaRepo.update(agenda.copyWith(syncStatus: SyncStatus.synced));
      } else {
        // 本地更新时间更晚，可能存在冲突
        await _handleConflict(localAgenda, agenda);
      }
    }
  }
  
  Future<void> _handleConflict(Agenda local, Agenda remote) async {
    // 简单策略：云端优先
    // 可扩展为：用户选择、字段合并等
    
    await localAgendaRepo.update(remote.copyWith(
      syncStatus: SyncStatus.synced,
    ));
    
    // 记录冲突日志
    await conflictLogRepo.insert(ConflictLog(
      entityType: 'agenda',
      entityId: local.id,
      localData: local.toJson(),
      remoteData: remote.toJson(),
      resolution: 'remote_priority',
    ));
  }
}
```

| 触发时机 | 同步内容 |
|---------|---------|
| **App启动** | 拉取云端最新数据（用户配置、事程、时间线） |
| **创建记录/事程后** | 立即推送新数据到云端 |

| **定时后台任务** | 每5分钟检查同步状态 |
