```dart
// lib/features/voice/services/voice_recording_service.dart

class VoiceRecordingService {
  final FlutterSoundRecorder _recorder = FlutterSoundRecorder();
  final VoiceWaveformController _waveformController;
  final OssUploadService _ossService;
  final AsrService _asrService;
  
  bool _isRecording = false;
  String? _currentRecordingPath;
  
  Future<void> initialize() async {
    await _recorder.openRecorder();
    await _recorder.setSubscriptionDuration(const Duration(milliseconds: 100));
  }
  
  /// 开始录音
  Future<void> startRecording() async {
    if (_isRecording) return;
    
    // 生成临时文件路径
    final directory = await getTemporaryDirectory();
    _currentRecordingPath = '${directory.path}/voice_${DateTime.now().millisecondsSinceEpoch}.wav';
    
    // 开始录音
    await _recorder.startRecorder(
      toFile: _currentRecordingPath,
      codec: Codec.pcm16WAV,
      numChannels: 1,
      sampleRate: 16000,  // 讯飞ASR要求16kHz
      bitRate: 256000,
    );
    
    _isRecording = true;
    
    // 监听录音数据，更新波形
    _recorder.onProgress!.listen((event) {
      // event.decibel 是当前音量（dB）
      _waveformController.updateWaveform(event.decibel ?? 0);
      
      // 检测是否静音（可选：自动停止）
      if (event.decibel != null && event.decibel! < -40) {
        // 静音检测
      }
    });
  }
  
  /// 结束录音
  Future<VoiceRecordingResult> stopRecording() async {
    if (!_isRecording) throw Exception('Not recording');
    
    final path = await _recorder.stopRecorder();
    _isRecording = false;
    
    // 获取录音时长
    final duration = await _getAudioDuration(path!);
    
    // 如果录音时长太短（<1秒），提示用户
    if (duration < 1.0) {
      return VoiceRecordingResult(
        success: false,
        error: '录音时长太短，请重新录制',
      );
    }
    
    return VoiceRecordingResult(
      success: true,
      filePath: path,
      duration: duration,
    );
  }
  
  /// 处理录音（上传+ASR+NLP）
  Future<ProcessedVoiceResult> processRecording(String filePath) async {
    // 1. 上传到OSS
    final ossUrl = await _ossService.uploadVoiceFile(filePath);
    
    // 2. 调用ASR
    final asrResult = await _asrService.recognize(filePath);
    
    if (!asrResult.success) {
      return ProcessedVoiceResult(
        success: false,
        error: '语音识别失败',
      );
    }
    
    // 3. 返回处理结果
    return ProcessedVoiceResult(
      success: true,
      ossUrl: ossUrl,
      text: asrResult.text,
      behaviorTag: asrResult.behaviorTag,
      nlpIntent: asrResult.intent,
      nlpTimeExpr: asrResult.timeExpr,
    );
  }
  
  Future<double> _getAudioDuration(String path) async {
    final player = FlutterSoundPlayer();
    await player.openPlayer();
    final duration = await player.startPlayer(fromURI: path);
    await player.stopPlayer();
    await player.closePlayer();
    return duration?.inMilliseconds / 1000.0 ?? 0;
  }
}

class VoiceWaveformController {
  final List<double> _waveformData = [];
  final int _maxDataPoints = 50;  // 显示最近50个数据点
  
  void updateWaveform(double decibel) {
    // 将dB转换为相对振幅（0-1）
    final amplitude = _decibelToAmplitude(decibel);
    
    _waveformData.add(amplitude);
    
    if (_waveformData.length > _maxDataPoints) {
      _waveformData.removeAt(0);
    }
  }
  
  double _decibelToAmplitude(double db) {
    // dB范围约 -160 到 0，映射到 0-1
    final normalized = (db + 160) / 160;
    return normalized.clamp(0.0, 1.0);
  }
  
  List<double> getWaveformData() => List.from(_waveformData);
}
```

```dart
// lib/features/voice/services/nlp_service.dart

class NlpService {
  final XunfeiNlpClient _xunfeiClient;
  final BehaviorTagRepository _behaviorTagRepo;
  final LocalNlpCache _localCache;
  
  /// 解析文本，提取行为信息
  Future<List<NlpParseResult>> parse(String text) async {
    // 1. 调用讯飞NLP API
    final nlpResponse = await _xunfeiClient.parse(text);
    
    // 2. 本地规则增强
    final results = await _enhanceWithLocalRules(text, nlpResponse);
    
    // 3. 匹配行为标签库
    for (final result in results) {
      result.behaviorTag = await _matchBehaviorTag(result.rawText);
    }
    
    return results;
  }
  
  /// 本地规则增强
  Future<List<NlpParseResult>> _enhanceWithLocalRules(
    String text, 
    NlpApiResponse nlpResponse,
  ) async {
    final results = <NlpParseResult>[];
    
    // 分句处理
    final sentences = _splitSentences(text);
    
    for (final sentence in sentences) {
      final result = NlpParseResult(
        rawText: sentence,
        intent: _detectIntent(sentence),
        timeExpr: _extractTimeExpr(sentence),
        statusExpr: _extractStatusExpr(sentence),
      );
      
      results.add(result);
    }
    
    return results;
  }
  
  /// 检测意图类型
  String _detectIntent(String text) {
    // 规则匹配
    if (_containsPlanKeywords(text)) {
      return 'plan';  // 计划意图："准备"、"打算"、"要"
    }
    if (_containsRecordKeywords(text)) {
      return 'record';  // 记录意图："正在"、"在"、"刚"
    }
    if (_containsQueryKeywords(text)) {
      return 'query';  // 查询意图："今天"、"做了什么"
    }
    
    // 默认为记录意图
    return 'record';
  }
  
  bool _containsPlanKeywords(String text) {
    const keywords = ['准备', '打算', '要', '想', '计划', '安排'];
    return keywords.any((k) => text.contains(k));
  }
  
  bool _containsRecordKeywords(String text) {
    const keywords = ['正在', '在', '刚', '已经', '完成', '做完'];
    return keywords.any((k) => text.contains(k));
  }
  
  bool _containsQueryKeywords(String text) {
    const keywords = ['今天', '做了什么', '记录', '查看'];
    return keywords.any((k) => text.contains(k));
  }
  
  /// 提取时间表达式
  String? _extractTimeExpr(String text) {
    // 时间正则匹配
    final timePatterns = [
      RegExp(r'(\d{1,2})点(\d{1,2})分?'),  // "9点30分"
      RegExp(r'(\d{1,2}):(\d{1,2})'),      // "9:30"
      RegExp(r'(早上|上午|中午|下午|晚上|睡前)'),  // 时间段
      RegExp(r'(早饭|午饭|晚饭)后'),       // 相对时间
      RegExp(r'(\d+)分钟后'),              // 相对时长
    ];
    
    for (final pattern in timePatterns) {
      final match = pattern.firstMatch(text);
      if (match != null) {
        return match.group(0);
      }
    }
    
    return null;
  }
  
  /// 提取状态表达式
  String? _extractStatusExpr(String text) {
    const statusKeywords = {
      '正在': 'in_progress',
      '在': 'in_progress',
      '刚': 'just_completed',
      '已经': 'completed',
      '完成': 'completed',
      '做完': 'completed',
    };
    
    for (final entry in statusKeywords.entries) {
      if (text.contains(entry.key)) {
        return entry.value;
      }
    }
    
    return null;
  }
  
  /// 匹配行为标签
  Future<String?> _matchBehaviorTag(String text) async {
    // 1. 从本地标签库查询
    final tags = await _behaviorTagRepo.getAllTags();
    
    for (final tag in tags) {
      for (final keyword in tag.matchKeywords) {
        if (text.contains(keyword)) {
          return tag.tagName;
        }
      }
    }
    
    // 2. 如果没有匹配，尝试提取动词+名词组合
    final verbNounPattern = RegExp(r'(吃|喝|拿|取|整理|收拾|运动|锻炼|休息|睡觉)(.+)');
    final match = verbNounPattern.firstMatch(text);
    if (match != null) {
      return '${match.group(1)}${match.group(2)}';
    }
    
    return null;
  }
  
  /// 分句处理
  List<String> _splitSentences(String text) {
    // 按逗号、顿号、空格分割
    return text.split(RegExp(r'[，、,，\s]+'))
        .where((s) => s.trim().length > 0)
        .toList();
  }
}
```

```dart
// lib/core/utils/time_parser.dart

class TimeParser {
  /// 解析时间表达式，返回DateTime
  Future<DateTime?> parseTimeExpr(String? timeExpr, DateTime referenceTime) async {
    if (timeExpr == null) return null;
    
    // 1. 绝对时间
    final absoluteTime = _parseAbsoluteTime(timeExpr, referenceTime);
    if (absoluteTime != null) return absoluteTime;
    
    // 2. 时间段
    final timeRange = _parseTimeRange(timeExpr);
    if (timeRange != null) {
      return _getDefaultTimeInRange(timeRange, referenceTime);
    }
    
    // 3. 相对时间（行为后）
    final relativeTime = await _parseRelativeTime(timeExpr, referenceTime);
    if (relativeTime != null) return relativeTime;
    
    // 4. 相对时长
    final relativeDuration = _parseRelativeDuration(timeExpr, referenceTime);
    if (relativeDuration != null) return relativeDuration;
    
    return null;
  }
  
  DateTime? _parseAbsoluteTime(String expr, DateTime reference) {
    // "9点30分" -> 09:30
    final pattern1 = RegExp(r'(\d{1,2})点(\d{1,2})分?');
    final match1 = pattern1.firstMatch(expr);
    if (match1 != null) {
      final hour = int.parse(match1.group(1)!);
      final minute = int.parse(match1.group(2)!);
      return DateTime(reference.year, reference.month, reference.day, hour, minute);
    }
    
    // "9:30" -> 09:30
    final pattern2 = RegExp(r'(\d{1,2}):(\d{1,2})');
    final match2 = pattern2.firstMatch(expr);
    if (match2 != null) {
      final hour = int.parse(match2.group(1)!);
      final minute = int.parse(match2.group(2)!);
      return DateTime(reference.year, reference.month, reference.day, hour, minute);
    }
    
    return null;
  }
  
  TimeRange? _parseTimeRange(String expr) {
    const ranges = {
      '早上': TimeRange(start: 6, end: 9),
      '上午': TimeRange(start: 9, end: 12),
      '中午': TimeRange(start: 12, end: 14),
      '下午': TimeRange(start: 14, end: 18),
      '晚上': TimeRange(start: 18, end: 22),
      '睡前': TimeRange(start: 21, end: 23),
    };
    
    return ranges[expr];
  }
  
  DateTime _getDefaultTimeInRange(TimeRange range, DateTime reference) {
    // 取时间范围的中间值
    final midHour = (range.start + range.end) / 2;
    return DateTime(reference.year, reference.month, reference.day, midHour.floor(), 0);
  }
```

```dart
// lib/features/match/domain/usecases/match_engine.dart

class MatchEngine {
  final AgendaRepository _agendaRepo;
  final TimelineRepository _timelineRepo;
  final MatchResultRepository _matchResultRepo;
  final BehaviorTagRepository _behaviorTagRepo;
  
  // 配置参数
  final double _keywordThreshold = 0.8;
  final int _timeWindowMinutes = 30;
  
  /// 执行匹配
  Future<MatchResult> match(TimelineRecord record) async {
    // 1. 查询待匹配事程
    final pendingAgendas = await _agendaRepo.getActiveAgendas(
      record.userId,
      record.timestamp,
    );
    
    if (pendingAgendas.isEmpty) {
      return MatchResult.noMatch();
    }
    
    // 2. 关键词匹配
    final keywordMatches = await _keywordMatch(record, pendingAgendas);
    
    if (keywordMatches.isEmpty) {
      return MatchResult.noMatch();
    }
    
    // 3. 时间窗口校验
    final timeWindowMatches = _timeWindowMatch(record, keywordMatches);
    
    if (timeWindowMatches.isEmpty) {
      return MatchResult.noMatch(reason: '时间窗口不匹配');
    }
    
    // 4. 选择最佳匹配
    final bestMatch = _selectBestMatch(timeWindowMatches);
    
    // 5. 更新状态
    await _updateMatchStatus(record, bestMatch);
    
    return MatchResult.success(
      agenda: bestMatch,
      matchScore: bestMatch.matchScore,
      timeDeviation: bestMatch.timeDeviation,
    );
  }
  
  /// 关键词匹配
  Future<List<AgendaMatchCandidate>> _keywordMatch(
    TimelineRecord record,
    List<Agenda> agendas,
  ) async {
    final candidates = <AgendaMatchCandidate>[];
    
    for (final agenda in agendas) {
      final score = await _calculateSimilarity(
        record.behaviorTag ?? '',
        agenda.behaviorTag ?? '',
      );
      
      if (score >= _keywordThreshold) {
        candidates.add(AgendaMatchCandidate(
          agenda: agenda,
          keywordMatchScore: score,
        ));
      }
    }
    
    return candidates;
  }
  
  /// 计算关键词相似度
  Future<double> _calculateSimilarity(String tag1, String tag2) async {
    // 1. 完全匹配
    if (tag1 == tag2) return 1.0;
    
    // 2. 从行为标签库获取匹配关键词
    final tag1Keywords = await _behaviorTagRepo.getMatchKeywords(tag1);
    final tag2Keywords = await _behaviorTagRepo.getMatchKeywords(tag2);
    
    // 3. 检查是否有交叉
    final intersection = tag1Keywords.intersection(tag2Keywords);
    if (intersection.isNotEmpty) {
      return 0.9;
    }
    
    // 4. 文本相似度（Jaccard）
    final words1 = tag1.split('');
    final words2 = tag2.split('');
    final set1 = Set.from(words1);
    final set2 = Set.from(words2);
    
    final intersectionSize = set1.intersection(set2).length;
    final unionSize = set1.union(set2).length;
    
    return intersectionSize / unionSize;
  }
  
  /// 时间窗口校验
  List<AgendaMatchCandidate> _timeWindowMatch(
    TimelineRecord record,
    List<AgendaMatchCandidate> candidates,
  ) {
    final validCandidates = <AgendaMatchCandidate>[];
    
    for (final candidate in candidates) {
      final deviation = record.timestamp.difference(candidate.agenda.plannedTime).inMinutes;
      
      // 时间偏差在允许范围内
      if (deviation.abs() <= _timeWindowMinutes) {
        validCandidates.add(candidate.copyWith(
          timeWindowMatch: true,
          timeDeviation: deviation,
        ));
      }
    }
    
    return validCandidates;
  }
  
  /// 选择最佳匹配
  AgendaMatchCandidate _selectBestMatch(List<AgendaMatchCandidate> candidates) {
    // 按匹配分数和时间偏差综合排序
    candidates.sort((a, b) {
      // 优先选择关键词匹配分数高的
      if (a.keywordMatchScore != b.keywordMatchScore) {
        return b.keywordMatchScore.compareTo(a.keywordMatchScore);
      }
      // 其次选择时间偏差小的
      return a.timeDeviation.abs().compareTo(b.timeDeviation.abs());
    });
    
    return candidates.first;
  }
  
  /// 更新匹配状态
  Future<void> _updateMatchStatus(
    TimelineRecord record,
    AgendaMatchCandidate candidate,
  ) async {
    // 计算综合匹配分数
    final matchScore = candidate.keywordMatchScore * 0.7 + 
        (1 - candidate.timeDeviation.abs() / _timeWindowMinutes) * 0.3;
    
    // 更新时间线记录
    await _timelineRepo.update(record.copyWith(
      matchedAgendaId: candidate.agenda.id,
      matchScore: matchScore,
      matchMethod: 'auto',
      matchedAt: DateTime.now(),
    ));
    
    // 更新事程状态
    await _agendaRepo.update(candidate.agenda.copyWith(
      status: AgendaStatus.matched,
      matchedTimelineId: record.id,
      matchedAt: DateTime.now(),
    ));
    
    // 记录匹配结果
    await _matchResultRepo.insert(MatchResultRecord(
      timelineId: record.id,
      agendaId: candidate.agenda.id,
      userId: record.userId,
      matchScore: matchScore,
      matchMethod: 'auto',
      keywordMatchScore: candidate.keywordMatchScore,
      timeWindowMatch: true,
      timeDeviationMinutes: candidate.timeDeviation,
      status: 'auto_matched',
    ));
  }
}

class AgendaMatchCandidate {
  final Agenda agenda;
  final double keywordMatchScore;
  final bool timeWindowMatch;
  final int timeDeviation;
  
  AgendaMatchCandidate({
    required this.agenda,
    required this.keywordMatchScore,
    this.timeWindowMatch = false,
    this.timeDeviation = 0,
  });
  
  double get matchScore => keywordMatchScore;
  
  AgendaMatchCandidate copyWith({
    bool? timeWindowMatch,
    int? timeDeviation,
  }) {
    return AgendaMatchCandidate(
      agenda: agenda,
      keywordMatchScore: keywordMatchScore,
      timeWindowMatch: timeWindowMatch ?? this.timeWindowMatch,
      timeDeviation: timeDeviation ?? this.timeDeviation,
    );
  }
}

class MatchResult {
  final bool success;
  final Agenda? agenda;
  final double? matchScore;
  final int? timeDeviation;
  final String? reason;
  
  MatchResult({
    required this.success,
    this.agenda,
    this.matchScore,
    this.timeDeviation,
    this.reason,
  });
  
  static MatchResult success({
    required Agenda agenda,
    required double matchScore,
    required int timeDeviation,
  }) {
    return MatchResult(
      success: true,
      agenda: agenda,
      matchScore: matchScore,
      timeDeviation: timeDeviation,
    );
  }
  
  static MatchResult noMatch({String? reason}) {
    return MatchResult(
      success: false,
      reason: reason ?? '无匹配事程',
    );
  }
}
```

```dart
  Future<SmartTimeResult?> _inferFromHistory(String behaviorTag, DateTime referenceDate) async {
    // 查询过去7-14天的该行为记录
    final historyRecords = await _timelineRepo.getRecordsByBehaviorTag(
      behaviorTag,
      startDate: referenceDate.subtract(Duration(days: 14)),
      endDate: referenceDate.subtract(Duration(days: 1)),
    );
    
    if (historyRecords.length < 5) {
      // 数据不足，无法推断
      return null;
    }
    
    // 计算平均时间
    final times = historyRecords.map((r) => r.timestamp).toList();
    final averageTime = _calculateMedianTime(times);
    
    // 计算时间一致性（标准差）
    final stdDeviation = _calculateStdDeviation(times);
    
    if (stdDeviation > 45) {
      // 时间波动太大，不适合推断
      return SmartTimeResult(
        layer: 1,
        suggestedTime: null,
        confidence: 0.5,
        reason: '历史时间波动较大（标准差${stdDeviation}分钟），建议手动设置',
        historicalData: SmartTimeHistoricalData(
          averageTime: averageTime,
          stdDeviation: stdDeviation,
          recordCount: historyRecords.length,
        ),
      );
    }
    
    // 推断成功
    final suggestedTime = DateTime(
      referenceDate.year,
      referenceDate.month,
      referenceDate.day,
      averageTime.hour,
      averageTime.minute,
    );
    
    return SmartTimeResult(
      layer: 1,
      suggestedTime: suggestedTime,
      confidence: 0.8 + (historyRecords.length / 14) * 0.1,  // 0.8-0.9
      reason: '根据您过去${historyRecords.length}天的习惯，平均在${_formatTime(averageTime)}${behaviorTag}',
      historicalData: SmartTimeHistoricalData(
        averageTime: averageTime,
        stdDeviation: stdDeviation,
        recordCount: historyRecords.length,
      ),
    );
  }
  
  /// 第三层：默认时段推荐
  Future<SmartTimeResult> _recommendDefaultTime(String behaviorTag) async {
    // 从行为标签库获取默认时间范围
    final tag = await _behaviorTagRepo.getByTagName(behaviorTag);
    
    if (tag != null) {
      final defaultStart = tag.defaultTimeRangeStart;
      final defaultEnd = tag.defaultTimeRangeEnd;
      
      // 取中间值
      final midHour = (defaultStart.hour + defaultEnd.hour) / 2;
      
      return SmartTimeResult(
        layer: 3,
        suggestedTime: null,  // 不直接给出时间，而是给出范围
        defaultRange: TimeRangeResult(
          start: defaultStart,
          end: defaultEnd,
          midTime: TimeOfDay(hour: midHour.floor(), minute: 0),
        ),
        confidence: 0.5,
        reason: '建议安排在${_formatTimeOfDay(defaultStart)}-${_formatTimeOfDay(defaultEnd)}之间',
      );
    }
    
    // 通用默认时段
    return SmartTimeResult(
      layer: 3,
      suggestedTime: null,
      defaultRange: TimeRangeResult(
        start: TimeOfDay(hour: 9, minute: 0),
        end: TimeOfDay(hour: 18, minute: 0),
        midTime: TimeOfDay(hour: 12, minute: 0),
      ),
      confidence: 0.3,
      reason: '建议安排在上午或下午',
    );
  }
  
  TimeOfDay _calculateMedianTime(List<DateTime> times) {
    // 提取小时和分钟
    final minutes = times.map((t) => t.hour * 60 + t.minute).toList();
    minutes.sort();
    
    final medianMinutes = minutes[minutes.length ~/ 2];
    return TimeOfDay(hour: medianMinutes ~/ 60, minute: medianMinutes % 60);
  }
  
  int _calculateStdDeviation(List<DateTime> times) {
    final minutes = times.map((t) => t.hour * 60 + t.minute).toList();
    final mean = minutes.reduce((a, b) => a + b) / minutes.length;
    
    final variance = minutes.map((m) => (m - mean) * (m - mean)).reduce((a, b) => a + b) / minutes.length;
    return sqrt(variance).round();
  }
  
  String _formatTime(TimeOfDay time) {
    return '${time.hour.toString().padLeft(2, '0')}:${time.minute.toString().padLeft(2, '0')}';
  }
  
  String _formatTimeOfDay(TimeOfDay time) {
    return '${time.hour.toString().padLeft(2, '0')}:${time.minute.toString().padLeft(2, '0')}';
  }
}
```

```dart
class QuestionClassifier {
  final NlpService _nlpService;
  final LocalRuleEngine _ruleEngine;
  
  /// 分类问题类型
  Future<QuestionClassification> classify(String question) async {
    // 1. 规则匹配（优先）
    final ruleResult = _matchByRules(question);
    if (ruleResult.confidence > 0.85) {
      return ruleResult;
    }
    
    // 2. 调用DeepSeek进行复杂意图识别
    final deepseekResult = await _nlpService.classifyIntent(question);
    
    // 3. 合并结果
    return _mergeResults(ruleResult, deepseekResult);

    // 物品位置规则
    if (_containsItemLocationKeywords(question)) {
      return QuestionClassification(
        type: QuestionType.itemLocation,
        confidence: 0.9,
        extractedItem: _extractItemName(question),
      );
    }
    
    // 时间查询规则
    if (_containsTimeQueryKeywords(question)) {
      return QuestionClassification(
        type: QuestionType.timeQuery,
        confidence: 0.85,
        timeRange: _parseTimeRange(question),
      );
    }
    
    // 频率统计规则

      return QuestionClassification(
        type: QuestionType.frequencyStats,
        confidence: 0.8,

      return QuestionClassification(
        type: QuestionType.comparisonAnalysis,
        confidence: 0.85,
        comparisonType: _extractComparisonType(question),
      );
    }
    
    // 默认：复杂拆解（调用DeepSeek处理）
```

```dart
// lib/features/qa/services/nlg_service.dart

class NlgService {
  final DeepSeekClient _deepseekClient;
  final TemplateEngine _templateEngine;
  final TtsService _ttsService;
  
  /// 生成回答
  Future<NlgResult> generateAnswer({
    required String question,
    required QuestionType questionType,
    required List<dynamic> searchResults,
    bool enableVoice = false,
  }) async {
    // 1. 简单问题使用模板生成（成本低、延迟低）
    if (_canUseTemplate(questionType)) {
      final templateAnswer = _templateEngine.generate(
        questionType: questionType,
        results: searchResults,
      );
      
      return NlgResult(
        text: templateAnswer,
        source: 'template',
        confidence: 0.9,
      );
    }
    
    // 2. 复杂问题调用DeepSeek生成自然语言回答
    final deepseekAnswer = await _deepseekClient.generateAnswer(
      question: question,
      context: _buildContext(searchResults),
      maxTokens: 500,
    );
    
    // 3. 可选：生成语音回答
    if (enableVoice) {
      final voiceUrl = await _ttsService.synthesize(deepseekAnswer.text);
      return NlgResult(
        text: deepseekAnswer.text,
        voiceUrl: voiceUrl,
        source: 'deepseek',
        confidence: deepseekAnswer.confidence,
      );
    }
    
    return NlgResult(
      text: deepseekAnswer.text,
      source: 'deepseek',
      confidence: deepseekAnswer.confidence,
    );
  }
  
  /// 判断是否可以使用模板
  bool _canUseTemplate(QuestionType type) {
    return type == QuestionType.itemLocation ||
           type == QuestionType.frequencyStats ||
           type == QuestionType.timeQuery;
  }
  
  /// 构建上下文
  String _buildContext(List<dynamic> results) {
    final contextBuilder = StringBuffer();
    
    for (final result in results) {
      if (result is TimelineRecord) {
        contextBuilder.writeln('- ${result.timestamp}: ${result.content}');
      } else if (result is ItemRecord) {
        contextBuilder.writeln('- ${result.itemName} 放在 ${result.location}');
      }
    }
    
    return contextBuilder.toString();
  }
}

// 模板引擎示例
class TemplateEngine {
  String generate({
    required QuestionType questionType,
    required List<dynamic> results,
  }) {
    switch (questionType) {
      case QuestionType.itemLocation:
        final item = results.first as ItemRecord;
        return '根据您 ${_formatDate(item.recordedAt)} 的记录，您把 ${item.itemName} 放在了 ${item.location}里。';
        
      case QuestionType.frequencyStats:
        final stats = results.first as FrequencyStats;
        return '这段时间您${stats.behaviorTag}了 ${stats.count} 次，平均每天 ${stats.dailyAverage.toStringAsFixed(1)} 次。';
        
      case QuestionType.timeQuery:
        final records = results as List<TimelineRecord>;
        final summary = records.map((r) => '${_formatTime(r.timestamp)} ${r.content}').join('\n');
        return '您在那天做了以下事情：\n$summary';
        
      default:
        return '';
    }
  }
  
  String _formatDate(DateTime date) {
    return '${date.month}月${date.day}日';
  }
  
  String _formatTime(DateTime time) {
    return '${time.hour}:${time.minute.toString().padLeft(2, '0')}';
  }
}
```

```dart
class DialogManager {
  final QaSessionRepository _sessionRepo;
  final QaHistoryRepository _historyRepo;
  
  /// 处理多轮对话
  Future<QaResult> handleQuestion({
    required String userId,
    required String question,
    String? sessionId,
  }) async {
    // 1. 获取或创建会话
    var session = sessionId != null
        ? await _sessionRepo.getById(sessionId)
        : await _createNewSession(userId);
    
    // 2. 加载历史上下文
    final history = await _historyRepo.getBySession(session.id);
    final context = _buildContextFromHistory(history);
    
    // 3. 判断是否追问（关联前文）
    if (_isFollowUpQuestion(question, context)) {
      return await _handleFollowUp(session, question, context);
    }
    
    // 4. 处理新问题
    return await _handleNewQuestion(session, question);
  }
  
  /// 判断是否追问
  bool _isFollowUpQuestion(String question, DialogContext context) {
    // 追问特征：包含"具体"、"详细"、"那"、"还有"等
    final followUpPatterns = ['具体', '详细', '那', '还有', '另外', '之后'];
    for (final pattern in followUpPatterns) {
      if (question.contains(pattern)) return true;
    }
    
    // 包含前文中的关键词
    if (context.lastQuestionType == QuestionType.timeQuery) {
      if (question.contains('几点') || question.contains('多久')) {
        return true;
      }
    }
    
    return false;
  }
  
  /// 处理追问
  Future<QaResult> _handleFollowUp(
    QaSession session,
    String question,
    DialogContext context,
  ) async {
    // 合并前文信息
    final enhancedQuestion = _enhanceWithContext(question, context);
    
    // 处理追问
    final result = await _processQuestion(enhancedQuestion);
    
    // 保存历史
    await _saveHistory(session.id, question, result);
    
    return result;
  }
  
  /// 合并上下文
  String _enhanceWithContext(String question, DialogContext context) {
    // 示例："具体几点？" -> "上周三吃药具体几点？"
    if (context.lastBehaviorTag != null) {
      return '${context.lastTimeRange ?? ''} ${context.lastBehaviorTag} $question';
    }
    return question;
  }
}
```

```go
func (s *QAService) ProcessQuestion(ctx context.Context, req *QuestionRequest) (*QuestionResponse, error) {
    // 1. 问题分类

    if err != nil {
        return nil, err
    }
    
    // 2. 多轮对话管理
    var dialogContext *DialogContext
    if req.SessionID != nil {
        dialogContext = s.dialogManager.GetContext(*req.SessionID)
    }

    if err != nil {

    }

    // 5. 自然语言生成回答

        return nil, err
    }
    
    // 6. 保存问答历史
    if req.SessionID != nil {
        s.dialogManager.SaveHistory(*req.SessionID, req.Question, answer)
    }
    
    // 7. 构建检索记录列表

// 执行检索
func (s *QAService) executeSearch(ctx context.Context, userID string, classification *QuestionClassification, context *DialogContext) ([]interface{}, error) {
    results := []interface{}{}
    
    switch classification.Type {
    case QuestionTypeItemLocation:
        // 物品位置查询
        itemResults, err := s.itemSearchEngine.Search(ctx, userID, classification.ExtractedItem)
        if err != nil {
            return nil, err
        }
        results = append(results, itemResults...)
        
    case QuestionTypeTimeQuery:
        // 时间查询
        timeRange := classification.TimeRange
        if context != nil && context.LastTimeRange != nil {
            timeRange = context.LastTimeRange
        }
        timelineResults, err := s.timelineSearchEngine.SearchByTimeRange(ctx, userID, timeRange)
        if err != nil {
            return nil, err
        }
        results = append(results, timelineResults...)
        
    case QuestionTypeFrequencyStats:
        // 频率统计
        stats, err := s.analyticsService.GetFrequencyStats(ctx, userID, classification.BehaviorTag, classification.TimeRange)
        if err != nil {
            return nil, err
        }
        results = append(results, stats)
        
    case QuestionTypePeriodReasoning:
        // 周期推理
        reasoning, err := s.knowledgeReasoner.Reason(ctx, userID, classification.ExtractedEntity)
        if err != nil {
            return nil, err
        }
        results = append(results, reasoning)
        
    case QuestionTypeComplexDecompose:
        // 复杂拆解：多路检索
        decomposedQuestions := s.questionClassifier.Decompose(classification.RawQuestion)
        for _, subQuestion := range decomposedQuestions {
            subClassification := s.questionClassifier.Classify(subQuestion)
            subResults, err := s.executeSearch(ctx, userID, subClassification, context)
            if err == nil {
                results = append(results, subResults...)
            }
        }
        
    case QuestionTypeComparisonAnalysis:
        // 对比分析
        comparison, err := s.analyticsService.ComparePeriods(ctx, userID, classification.Period1, classification.Period2, classification.BehaviorTag)
        if err != nil {
            return nil, err
        }
        results = append(results, comparison)
        
    case QuestionTypePatternDiscovery:
        // 寻找规律
        patterns, err := s.analyticsService.DiscoverPatterns(ctx, userID, classification.BehaviorTag)
        if err != nil {
            return nil, err
        }
        results = append(results, patterns)
    }
    
    return results, nil
}
```

```dart
  /// 获取记录类型的显示名称
  String get recordTypeLabel {
    switch (recordType) {
      case 'timeline':
        return '时间线记录';
      case 'item':
        return '物品位置';
      case 'knowledge':
        return '知识条目';
      case 'agenda':
        return '计划事程';
      default:
        return '记忆记录';
    }
  }

  /// 获取相关性评分颜色
  Color get relevanceColor {
    if (relevanceScore >= 0.8) return Color(0xFF6B9E75);
    if (relevanceScore >= 0.5) return Color(0xFFD4A574);
    return Color(0xFF8A8A8A);
  }

  /// 格式化时间戳
  String get formattedTimestamp {
    return '${timestamp.month}月${timestamp.day}日 ${timestamp.hour.toString().padLeft(2, '0')}:${timestamp.minute.toString().padLeft(2, '0')}';
  }
}

/// 回答检索记录展示组件

class AnswerSourcesWidget extends StatefulWidget {
  final List<QaSourceRecord> sources;
  final VoidCallback? onCopyAll;

  const AnswerSourcesWidget({
    required this.sources,
    this.onCopyAll,
  });

  @override

}

class _AnswerSourcesWidgetState extends State<AnswerSourcesWidget>
    with SingleTickerProviderStateMixin {
  bool _isExpanded = false;
  late AnimationController _animationController;
  late Animation<double> _rotationAnimation;

  @override
  void initState() {
    super.initState();
    _animationController = AnimationController(
      duration: Duration(milliseconds: 200),
      vsync: this,
    );
    _rotationAnimation = Tween<double>(
      begin: 0,
      end: 0.5,
    ).animate(_animationController);
  }

  @override
  void dispose() {
    _animationController.dispose();
    super.dispose();
  }

  void _toggleExpanded() {
    setState(() {
      _isExpanded = !_isExpanded;
      if (_isExpanded) {
        _animationController.forward();
      } else {
        _animationController.reverse();
      }
    });
  }

  /// 复制单条记录到剪贴板
  Future<void> _copySingleRecord(QaSourceRecord record) async {
    final text = '${record.recordTypeLabel} | ${record.formattedTimestamp}\n'
        '内容：${record.content}\n'
        '相关性：${(record.relevanceScore * 100).toStringAsFixed(1)}%';
    
    await Clipboard.setData(ClipboardData(text: text));
    
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text('已复制到剪贴板'),
        duration: Duration(seconds: 1),
        behavior: SnackBarBehavior.floating,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(8)),
      ),
    );
  }

    final buffer = StringBuffer();
    buffer.writeln('=== 相关记忆记录 (${widget.sources.length}条) ===');
    buffer.writeln();
    
    for (int i = 0; i < widget.sources.length; i++) {
      final record = widget.sources[i];
      buffer.writeln('[$i] ${record.recordTypeLabel}');
      buffer.writeln('时间：${record.formattedTimestamp}');
      buffer.writeln('内容：${record.content}');
      buffer.writeln('相关性：${(record.relevanceScore * 100).toStringAsFixed(1)}%');
      if (record.behaviorTag != null) {
        buffer.writeln('标签：${record.behaviorTag}');
      }
      buffer.writeln();
    }
    
    await Clipboard.setData(ClipboardData(text: buffer.toString()));
    
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text('已复制全部 ${widget.sources.length} 条记录'),
        duration: Duration(seconds: 1),
        behavior: SnackBarBehavior.floating,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(8)),

      _messages.add(QaMessage(
        content: question,
        isUser: true,
        timestamp: DateTime.now(),

      _isProcessing = true;
    });
    
    _scrollToBottom();

      final response = await _qaService.askQuestion(
        question: question,
        sessionId: _sessionId,
      );
      
      // 更新会话ID
      if (_sessionId == null) {
        _sessionId = response.sessionId;

      // 添加AI回答（包含检索记录）

        _messages.add(QaMessage(
          content: response.answer,
          isUser: false,
          timestamp: DateTime.now(),
          sourceRecords: response.sourceRecords,
        ));
        _isProcessing = false;
      });
      
      _scrollToBottom();
      
      // 如果有语音回答，播放语音
      if (response.answerVoiceUrl != null) {
        await _playVoice(response.answerVoiceUrl!);
      }
    } catch (e) {
      setState(() {
        _messages.add(QaMessage(
          content: '抱歉，处理问题时出现错误。',
          isUser: false,
          timestamp: DateTime.now(),
        ));
        _isProcessing = false;
      });
    }

  factory QaResponse.fromJson(Map<String, dynamic> json) {
    return QaResponse(
      answer: json['answer'] as String,
      answerVoiceUrl: json['answerVoiceUrl'] as String?,
      confidence: (json['confidence'] as num).toDouble(),
      sessionId: json['sessionId'] as String?,
      sourceRecords: (json['sourceRecords'] as List<dynamic>?)
          ?.map((e) => QaSourceRecord.fromJson(e as Map<String, dynamic>))
          .toList(),
    );
  }
}
```

### 8.4 性能优化与缓存策略

| 场景             | 优化策略               | 技术实现                       |
| ---------------- | ---------------------- | ------------------------------ |
| **高频问题缓存** | 物品位置查询结果缓存   | Redis缓存，TTL=1小时           |
| **向量嵌入缓存** | 文本向量计算结果缓存   | Redis缓存向量，避免重复计算    |
| **检索结果缓存** | 常见查询结果缓存       | Redis缓存检索结果，TTL=30分钟  |
| **异步向量同步** | 时间线记录异步生成向量 | Kafka消息队列，异步处理        |
| **向量预热**     | 用户活跃时段预加载向量 | 后台任务，提前加载热门记录向量 |

```go
// 缓存策略实现示例

type QACacheService struct {
    redisClient *redis.Client
}

func (c *QACacheService) GetCachedItemLocation(userID, itemName string) (*ItemRecord, error) {
    key := fmt.Sprintf("qa:item:%s:%s", userID, itemName)
    cached, err := c.redisClient.Get(key).Bytes()
    if err == nil {
        var item ItemRecord
        json.Unmarshal(cached, &item)
        return &item, nil
    }
    return nil, err
}

func (c *QACacheService) CacheItemLocation(userID, itemName string, item *ItemRecord) error {
    key := fmt.Sprintf("qa:item:%s:%s", userID, itemName)
    data, _ := json.Marshal(item)
    return c.redisClient.Set(key, data, 1*time.Hour).Err()
}
```
